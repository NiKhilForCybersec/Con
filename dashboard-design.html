Executive Dashboard Elements

SPL: Executive Risk Summary
// Overall security posture
| stats count as total_incidents by severity
| eval severity_order = case(
    severity="Critical", 1,
    severity="High", 2,
    severity="Medium", 3,
    severity="Low", 4,
    5
)
| sort severity_order
| appendpipe [
    stats sum(total_incidents) as total_incidents
    | eval severity="TOTAL"
]
KQL: Monthly Trend Visualization
SecurityIncident
| where TimeGenerated > ago(90d)
| summarize IncidentCount = count() by 
    bin(TimeGenerated, 1d), Severity
| render timechart with (
    title="Daily Incident Trend by Severity"
)
Executive Dashboard Best Practices
  • Single page: Executives won't scrollâ€”everything on one screen
  • Traffic lights: Red/Yellow/Green status indicators at a glance
  • Trends over absolutes: "Incidents down 20%" beats "47 incidents"
  • Comparison periods: This month vs last month, this quarter vs last
  • Business context: Link security metrics to business impact

SOC Operations Dashboard

SPL: Alert Queue Health
index=alerts status=open
| eval age_hours = round((now() - _time) / 3600, 1)
| eval age_bucket = case(
    age_hours < 1, "< 1 hour",
    age_hours < 4, "1-4 hours",
    age_hours < 24, "4-24 hours",
    "Over 24 hours"
)
| stats count by age_bucket, severity
| xyseries age_bucket, severity, count
KQL: Analyst Workload Distribution
SecurityAlert
| where TimeGenerated > ago(7d)
| where Status == "InProgress"
| summarize 
    OpenAlerts = count(),
    CriticalCount = countif(Severity == "High"),
    AvgAgeHours = avg(datetime_diff('hour', now(), TimeGenerated))
  by AssignedTo = Owner
| extend WorkloadStatus = case(
    OpenAlerts > 50, "ğŸ”´ Overloaded",
    OpenAlerts > 30, "ğŸŸ¡ Heavy",
    "ğŸŸ¢ Normal"
)
| order by OpenAlerts desc
SPL: MTTD/MTTR Tracking
index=incidents status=closed
| eval time_to_detect = detected_time - event_time
| eval time_to_respond = closed_time - detected_time
| stats 
    avg(time_to_detect) as avg_mttd,
    avg(time_to_respond) as avg_mttr,
    perc90(time_to_detect) as p90_mttd,
    perc90(time_to_respond) as p90_mttr
  by severity
| eval avg_mttd_hours = round(avg_mttd / 3600, 1)
| eval avg_mttr_hours = round(avg_mttr / 3600, 1)

Analyst Investigation Dashboard

KQL: Alert Context Panel
// Single alert with full context
let AlertId = "ALERT-12345";
let AlertData = SecurityAlert | where SystemAlertId == AlertId;
let TargetIP = toscalar(AlertData | extend IP = tostring(parse_json(ExtendedProperties).["Destination IP"]) | project IP);
let TargetUser = toscalar(AlertData | extend User = tostring(parse_json(ExtendedProperties).User) | project User);
// Alert details
AlertData
| project TimeGenerated, AlertName, Description, Severity,
          Tactics, Techniques
| extend Section = "Alert Details"
| union (
    // Related alerts for same entity
    SecurityAlert
    | where TimeGenerated > ago(7d)
    | where ExtendedProperties has TargetIP or ExtendedProperties has TargetUser
    | summarize RelatedAlerts = count(), AlertTypes = make_set(AlertName)
    | extend Section = "Related Alerts"
)
| union (
    // Asset information
    _GetWatchlist('AssetInventory')
    | where IPAddress == TargetIP
    | project Hostname, Owner, Criticality, Department
    | extend Section = "Asset Info"
)

Detection Coverage Dashboard

SPL: MITRE ATT&CK Coverage
| inputlookup detection_rules.csv
| stats count as rule_count by mitre_tactic, mitre_technique
| join type=left mitre_technique [
    | inputlookup mitre_techniques.csv
    | table technique_id, technique_name
    | rename technique_id as mitre_technique
]
| eval coverage_status = case(
    rule_count >= 3, "Good",
    rule_count >= 1, "Partial",
    "Gap"
)
| table mitre_tactic, mitre_technique, technique_name, rule_count, coverage_status
KQL: Rule Performance Analysis
SecurityAlert
| where TimeGenerated > ago(30d)
| summarize 
    TotalFires = count(),
    TruePositives = countif(Status == "Resolved" and Classification == "TruePositive"),
    FalsePositives = countif(Classification == "FalsePositive")
  by AlertName = ProductName
| extend TPRate = round(TruePositives * 100.0 / TotalFires, 1)
| extend FPRate = round(FalsePositives * 100.0 / TotalFires, 1)
| extend Health = case(
    TPRate >= 80, "ğŸŸ¢ Healthy",
    TPRate >= 50, "ğŸŸ¡ Needs Tuning",
    "ğŸ”´ Review"
)
| order by TotalFires desc

Design Principles

Principle Good Practice Anti-Pattern
Actionability Each panel drives a decision or action Vanity metrics that look impressive but aren't useful
Context Show thresholds, baselines, comparisons Raw numbers without reference points
Hierarchy Most important info top-left, largest All panels same size, random placement
Drill-down Click panel to see underlying data Dead-end visualizations
Performance Panels load in < 5 seconds Heavy queries that timeout
ğŸ’¼ Interview Tip

Q: "How do you design effective security dashboards?"

A: "I start with the audience and their decisions:

  1. Know the audience: Executives need risk posture in 30 seconds. Analysts need actionable queues. Same data, different presentations.
  2. Action-oriented: Every panel should answer 'so what?' If it doesn't drive a decision, remove it. A panel showing total events is useless; one showing events exceeding baseline is actionable.
  3. Context is king: Numbers without context are meaningless. '500 alerts' means nothing. '500 alerts vs 200 baseline' tells a story. I always include thresholds, trends, or comparisons.
  4. Performance matters: A dashboard that takes 60 seconds to load won't be used. I optimize queries, use summary indexes, and cache where possible.
  5. Iterate with users: I deploy a draft, watch how people actually use it, then refine. Often what I think is important isn't what analysts need day-to-day.

The best dashboard is one people actually look at every day because it helps them do their job better."
