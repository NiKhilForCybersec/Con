Conditional Access Monitoring

KQL: Conditional Access Policy Evaluation
SigninLogs
| where TimeGenerated > ago(24h)
| mv-expand ConditionalAccessPolicies
| extend PolicyName = tostring(ConditionalAccessPolicies.displayName)
| extend PolicyResult = tostring(ConditionalAccessPolicies.result)
| summarize 
    TotalEvaluations = count(),
    Successes = countif(PolicyResult == "success"),
    Failures = countif(PolicyResult == "failure"),
    NotApplied = countif(PolicyResult == "notApplied")
  by PolicyName
| extend FailureRate = round(Failures * 100.0 / TotalEvaluations, 2)
| order by Failures desc
KQL: CA Policy Bypass Detection
// Detect sign-ins that bypassed expected policies
SigninLogs
| where TimeGenerated > ago(24h)
| where ResultType == 0  // Successful sign-in
| mv-expand ConditionalAccessPolicies
| where ConditionalAccessPolicies.result == "notApplied"
| extend PolicyName = tostring(ConditionalAccessPolicies.displayName)
| where PolicyName has_any ("MFA", "Compliant", "Block")
| project TimeGenerated, UserPrincipalName, AppDisplayName, 
          IPAddress, PolicyName
| summarize BypassedPolicies = make_set(PolicyName) by UserPrincipalName
| where array_length(BypassedPolicies) > 0

Device Compliance Monitoring

KQL: Non-Compliant Device Access
SigninLogs
| where TimeGenerated > ago(24h)
| where ResultType == 0
| extend DeviceCompliance = tostring(DeviceDetail.isCompliant)
| extend DeviceManaged = tostring(DeviceDetail.isManaged)
| where DeviceCompliance != "true" or DeviceManaged != "true"
| summarize 
    AccessCount = count(),
    Apps = make_set(AppDisplayName)
  by UserPrincipalName, DeviceCompliance, DeviceManaged
| order by AccessCount desc
SPL: Device Health Posture
index=endpoint_mgmt 
| eval compliance_status = case(
    antivirus_enabled="true" AND os_patched="true" AND encrypted="true", "Compliant",
    antivirus_enabled="false", "Non-Compliant: No AV",
    os_patched="false", "Non-Compliant: Unpatched",
    encrypted="false", "Non-Compliant: Unencrypted",
    "Unknown"
)
| stats count by compliance_status, device_type
| sort -count

Network Micro-Segmentation

SPL: Segment Violation Detection
index=firewall action=blocked
| lookup network_segments src_ip OUTPUT src_segment
| lookup network_segments dest_ip OUTPUT dest_segment
| where src_segment != dest_segment
| stats count as violations,
        values(dest_port) as ports_attempted
  by src_segment, dest_segment, src_ip
| where violations > 10
| sort -violations
KQL: East-West Traffic Anomalies
CommonSecurityLog
| where TimeGenerated > ago(24h)
| where SourceIP startswith "10." and DestinationIP startswith "10."
| where DeviceAction == "allow"
| summarize 
    Connections = count(),
    Destinations = dcount(DestinationIP),
    Ports = make_set(DestinationPort)
  by SourceIP
| where Destinations > 20  // Talking to many internal hosts
| order by Destinations desc

ZTNA/SASE Monitoring

SPL: ZTNA Access Patterns
index=ztna_proxy
| stats count as access_count,
        dc(application) as unique_apps,
        values(application) as apps_accessed,
        dc(src_ip) as locations
  by user
| where locations > 3  // Accessing from many locations
| sort -locations
SPL: Application Access Without ZTNA
// Detect direct application access bypassing ZTNA
index=app_access NOT [search index=ztna_proxy | fields user, application]
| eval bypass_indicator = "Direct access detected"
| stats count by user, application, src_ip
| sort -count

Continuous Risk Assessment

KQL: Real-Time User Risk
// Combine multiple risk signals
let RiskySignIns = SigninLogs
| where TimeGenerated > ago(1h)
| where RiskLevelDuringSignIn != "none"
| summarize SignInRisk = countif(RiskLevelDuringSignIn in ("medium", "high")) 
  by UserPrincipalName;
let PolicyViolations = SigninLogs
| where TimeGenerated > ago(1h)
| mv-expand ConditionalAccessPolicies
| where ConditionalAccessPolicies.result == "failure"
| summarize PolicyFailures = count() by UserPrincipalName;
let AnomalousAccess = SigninLogs
| where TimeGenerated > ago(1h)
| where ResultType == 0
| summarize UniqueLocations = dcount(Location) by UserPrincipalName
| where UniqueLocations > 2;
RiskySignIns
| join kind=leftouter PolicyViolations on UserPrincipalName
| join kind=leftouter AnomalousAccess on UserPrincipalName
| extend CompositeRisk = SignInRisk + coalesce(PolicyFailures, 0) * 2 + 
                          iff(isnotnull(UniqueLocations), 5, 0)
| where CompositeRisk > 5
| order by CompositeRisk desc

Q: "How does SIEM support Zero Trust architecture?"

A: "SIEM is the verification and visibility layer for Zero Trust:

  1. Policy enforcement validation: I monitor Conditional Access logs to ensure policies are applied. Bypasses or exceptions are immediately visible.
  2. Continuous risk assessment: Zero Trust means continuous verification. I aggregate risk signalsâ€”risky sign-ins, device posture, behavior anomaliesâ€”into real-time risk scores.
  3. Micro-segmentation monitoring: Network segmentation is only effective if enforced. I monitor firewall logs for segment violations and unauthorized lateral movement.
  4. Device compliance tracking: Access from non-compliant or unmanaged devices is a Zero Trust violation. I correlate device state with access events.
  5. Least privilege validation: I track permission changes and access patterns. Users accessing resources outside their normal pattern trigger investigation.
  6. Assume breach detection: Zero Trust assumes breach. SIEM detects post-compromise activityâ€”lateral movement, privilege escalation, data exfiltration.

Zero Trust without visibility is just policy. SIEM proves the policies are working."
