Sensitivity Labels in SIEM

KQL: Monitor Labeled Data Access
// Track access to classified data
OfficeActivity
| where TimeGenerated > ago(24h)
| where OfficeWorkload in ("SharePoint", "OneDrive")
| where Operation in ("FileAccessed", "FileDownloaded", "FileCopied")
| extend SensitivityLabel = tostring(SensitivityLabelId)
| where isnotempty(SensitivityLabel)
| lookup kind=leftouter (
    externaldata(LabelId:string, LabelName:string, Priority:int)
    [@"https://storage/sensitivity_labels.csv"]
) on $left.SensitivityLabel == $right.LabelId
| where Priority >= 3  // Confidential and above
| summarize 
    AccessCount = count(),
    UniqueFiles = dcount(SourceFileName),
    Operations = make_set(Operation)
  by UserId, LabelName, bin(TimeGenerated, 1h)
| where AccessCount > 20
| order by Priority desc, AccessCount desc
KQL: External Sharing of Sensitive Data
OfficeActivity
| where TimeGenerated > ago(7d)
| where Operation in ("SharingSet", "AnonymousLinkCreated")
| where TargetUserOrGroupType == "Guest" or Operation == "AnonymousLinkCreated"
| extend SensitivityLabel = tostring(SensitivityLabelId)
| lookup kind=leftouter (
    externaldata(LabelId:string, LabelName:string, AllowExternal:bool)
    [@"https://storage/sensitivity_labels.csv"]
) on $left.SensitivityLabel == $right.LabelId
| where AllowExternal == false  // External sharing not allowed
| project TimeGenerated, UserId, SourceFileName, LabelName, 
          SharedWith = TargetUserOrGroupName, Operation

Classification-Based Alerting

SPL: Risk-Based Alert Severity
index=dlp_events
| lookup data_classification file_path OUTPUT classification, risk_score
| eval alert_severity = case(
    classification="Restricted" AND action="blocked", "Critical",
    classification="Restricted" AND action="allowed", "Critical",
    classification="Confidential" AND action="blocked", "High",
    classification="Confidential" AND action="allowed", "High",
    classification="Internal" AND action="blocked", "Medium",
    "Low"
)
| where alert_severity IN ("Critical", "High")
| table _time, user, file_path, classification, action, alert_severity, destination
KQL: Asset Criticality Correlation
// Correlate alerts with data classification
let DataAssets = externaldata(AssetId:string, Classification:string, Owner:string)
[@"https://storage/asset_classification.csv"];
SecurityAlert
| where TimeGenerated > ago(24h)
| extend AffectedAssets = parse_json(Entities)
| mv-expand Asset = AffectedAssets
| where Asset.Type == "Host" or Asset.Type == "File"
| extend AssetId = tostring(Asset.Name)
| lookup kind=leftouter DataAssets on AssetId
| extend AdjustedSeverity = case(
    Classification == "Restricted", "Critical",
    Classification == "Confidential" and Severity == "Medium", "High",
    Severity
)
| project TimeGenerated, AlertName, AssetId, Classification, 
          OriginalSeverity = Severity, AdjustedSeverity, Owner

Data Discovery

SPL: Find Unclassified Sensitive Data
index=file_access file_extension IN ("xlsx", "csv", "docx", "pdf")
| where NOT [| inputlookup classified_files | fields file_path]
| rex field=file_content "(?\d{3}-\d{2}-\d{4})"
| rex field=file_content "(?\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4})"
| rex field=file_content "(?\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b)"
| where isnotnull(ssn) OR isnotnull(cc) OR isnotnull(email)
| stats count as sensitive_matches,
        values(ssn) as ssn_found,
        values(cc) as cc_found
  by file_path, file_owner
| where sensitive_matches > 0

Q: "How does data classification improve security monitoring?"

A: "Classification enables risk-based security operations:

  1. Alert prioritization: When I know data sensitivity, I can adjust alert severity. Access to public data is low priority; access to restricted data is critical.
  2. Detection tuning: I apply stricter detection rules to sensitive data stores. Bulk downloads from a restricted database warrant investigation; from marketing materials, they don't.
  3. DLP integration: Classification labels feed directly into DLP policies. SIEM ingests DLP events with classification context, enabling meaningful alerts.
  4. Incident response: When responding to incidents, knowing data classification helps assess impact. Breach of restricted data is reportable; internal data may not be.
  5. Compliance mapping: Different data types have different regulatory requirements. Classification enables compliance-specific monitoring and reporting.
  6. Resource allocation: Security teams have limited capacity. Classification ensures effort focuses on protecting what matters most.

Without classification, all data gets equal protectionâ€”which means important data is under-protected."
