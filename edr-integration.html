Microsoft Defender for Endpoint

KQL: MDE Alert Correlation
// Correlate MDE alerts with sign-in events
AlertInfo
| where TimeGenerated > ago(24h)
| where ServiceSource == "Microsoft Defender for Endpoint"
| join kind=inner (
    AlertEvidence
    | where EntityType == "User"
    | project AlertId, AccountName, AccountDomain
) on AlertId
| join kind=leftouter (
    SigninLogs
    | where TimeGenerated > ago(24h)
    | project UserPrincipalName, IPAddress, Location, 
              AppDisplayName, ResultType
) on $left.AccountName == $right.UserPrincipalName
| project TimeGenerated, Title, Severity, AccountName,
          IPAddress, Location, AppDisplayName
KQL: Process Chain Analysis
DeviceProcessEvents
| where TimeGenerated > ago(24h)
| where InitiatingProcessFileName in~ ("cmd.exe", "powershell.exe", "wscript.exe")
| where FileName in~ ("whoami.exe", "net.exe", "nltest.exe", "dsquery.exe")
| extend ProcessChain = strcat(
    InitiatingProcessParentFileName, " â†’ ",
    InitiatingProcessFileName, " â†’ ",
    FileName
)
| summarize 
    CommandCount = count(),
    Commands = make_set(ProcessCommandLine, 5)
  by DeviceName, AccountName, ProcessChain
| where CommandCount >= 3
| order by CommandCount desc

CrowdStrike Falcon

SPL: CrowdStrike Detection Analysis
index=crowdstrike sourcetype=crowdstrike:events:sensor
| where event_simpleName IN ("ProcessRollup2", "DnsRequest", "NetworkConnectIP4")
| stats count by ComputerName, event_simpleName, UserName
| where count > 100
| sort -count
SPL: Falcon Alerts to Incident
index=crowdstrike sourcetype=crowdstrike:events:detect
| eval severity_num = case(
    Severity="Critical", 4,
    Severity="High", 3,
    Severity="Medium", 2,
    Severity="Low", 1,
    0
)
| stats max(severity_num) as max_severity,
        values(Technique) as techniques,
        values(Tactic) as tactics,
        dc(ComputerName) as affected_hosts
  by DetectName
| where max_severity >= 3
| sort -max_severity

Cortex XDR

SPL: Cortex XDR Incident Enrichment
index=cortex_xdr sourcetype=pan:xdr:incident
| lookup asset_inventory hostname as host OUTPUT criticality, owner, department
| lookup user_risk user OUTPUT risk_score
| eval priority = case(
    severity="high" AND criticality="Critical", "P1",
    severity="high" OR criticality="Critical", "P2",
    severity="medium", "P3",
    "P4"
)
| table _time, incident_id, host, user, severity, 
        criticality, risk_score, priority, description
SPL: XDR + Network Correlation
index=cortex_xdr sourcetype=pan:xdr:alert alert_source="XDR Agent"
| join type=left host [
    search index=pan_firewall action=allowed
    | stats sum(bytes_sent) as bytes_out, 
            dc(dest_ip) as unique_dests 
      by src_ip
    | rename src_ip as host_ip
]
| where bytes_out > 100000000 OR unique_dests > 50
| table _time, host, alert_name, bytes_out, unique_dests

Cross-EDR Normalization

SPL: Unified EDR Alert View
// Normalize alerts from multiple EDR platforms
(index=crowdstrike sourcetype=crowdstrike:events:detect)
OR (index=cortex_xdr sourcetype=pan:xdr:alert)
OR (index=carbonblack sourcetype=carbonblack:defense:json type="ALERT")
| eval normalized_severity = case(
    // CrowdStrike
    Severity="Critical", "Critical",
    Severity="High", "High",
    // Cortex XDR
    severity="high", "High",
    severity="critical", "Critical",
    // Carbon Black
    threat_score >= 8, "Critical",
    threat_score >= 6, "High",
    "Medium"
)
| eval normalized_host = coalesce(ComputerName, host, device_name)
| eval normalized_user = coalesce(UserName, user, device_username)
| eval source_edr = case(
    isnotnull(ComputerName), "CrowdStrike",
    isnotnull(host), "Cortex XDR",
    isnotnull(device_name), "Carbon Black",
    "Unknown"
)
| table _time, source_edr, normalized_host, normalized_user, 
        normalized_severity, description
